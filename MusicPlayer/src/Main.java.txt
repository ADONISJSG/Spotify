import javafx.application.Application;
import javafx.geometry.Insets;
import javafx.geometry.Pos;
import javafx.scene.Scene;
import javafx.scene.control.*;
import javafx.scene.layout.*;
import javafx.stage.Stage;
import java.util.ArrayList;

public class Main extends Application {

    @Override
    public void start(Stage stage) {

        ArrayList<Song> songs = new ArrayList<>();
        songs.add(new Song("Song 1", "music/song1.mp3"));
        songs.add(new Song("Song 2", "music/song2.mp3"));
        songs.add(new Song("Song 3", "music/song2.mp3"));

        PlayerController controller = new PlayerController(songs);

        Label nowPlaying = new Label("Reproduciendo: " + controller.getCurrentSong().getTitle());
        nowPlaying.setStyle("-fx-text-fill: white; -fx-font-size: 14px;");

        Button play = new Button("▶");
        Button pause = new Button("⏸");
        Button next = new Button("⏭");
        Button prev = new Button("⏮");

        play.setOnAction(e -> {
            controller.play();
            nowPlaying.setText("Reproduciendo: " + controller.getCurrentSong().getTitle());
        });

        pause.setOnAction(e -> controller.pause());

        next.setOnAction(e -> {
            controller.next();
            nowPlaying.setText("Reproduciendo: " + controller.getCurrentSong().getTitle());
        });

        prev.setOnAction(e -> {
            controller.previous();
            nowPlaying.setText("Reproduciendo: " + controller.getCurrentSong().getTitle());
        });

        Slider volume = new Slider(0, 1, 0.5);
        volume.valueProperty().addListener((obs, oldVal, newVal) ->
                controller.setVolume(newVal.doubleValue())
        );

        HBox controls = new HBox(10, prev, play, pause, next);
        controls.setAlignment(Pos.CENTER);

        VBox root = new VBox(15, nowPlaying, controls, new Label("Volumen"), volume);
        root.setAlignment(Pos.CENTER);
        root.setPadding(new Insets(20));
        root.setStyle("-fx-background-color: #121212;");

        Scene scene = new Scene(root, 400, 300);
        stage.setTitle("Mini Spotify");
        stage.setScene(scene);
        stage.show();
    }

    public static void main(String[] args) {
        launch(args);
    }
}
